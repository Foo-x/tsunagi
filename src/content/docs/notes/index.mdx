import { getCollection } from "astro:content"

export const notes = (await getCollection("docs", ({ id }) => id.startsWith("notes/")))
  .sort((a, b) => a.data.title.localeCompare(b.data.title, "ja"))
  .map((note) => (
    <>
      <a href={`${import.meta.env.BASE_URL}/${note.id}`}>{note.data.title}</a>
      <br />
    </>
  ))

{notes}
